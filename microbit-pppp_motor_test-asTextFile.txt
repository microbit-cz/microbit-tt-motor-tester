run3 = 0

def Block(speed, pause, number, index):
    print("block func")
    
    Speed(speed)
    basic.pause(pause)
    TurnsCount(number , index)

def Speed(speed):
    print("speed func")
    PCAmotor.motor_run(PCAmotor.Motors.M2, speed)

def TurnsCount(x, index):
    global run1, run2, run3
    print("turnscount func")
    turnscount = 0
    y = 0
    hole = False;
    while (x != y):
        
        if pins.analog_read_pin(AnalogPin.P0) > 100:
            if hole == False:
                turnscount += 1
                hole = True
        if pins.analog_read_pin(AnalogPin.P0) < 100:
            hole = False
        
        basic.pause(.5)
        y +=1
    if index == 0: run1 = turnscount
    elif index == 1: run2 =turnscount
    else: run3 = turnscount
    



    

def Grade():
    PCAmotor.motor_run(PCAmotor.Motors.M3, 0)
    global run1, run2, run3
    print("grade func")
    print(run1)
    print(run2)
    print(run3)
    AVG = (run1+run2+run3)/3
    base = 135
    grade = 0
    if AVG > base + 50:
        grade = 10
    elif AVG >= base + 40:
        grade = 9
    elif AVG >= base + 30:
        grade = 8
    elif AVG >= base + 20:
        grade = 7
    elif AVG >= base + 10:
        grade = 6
    elif AVG >= base:
        grade = 5
    elif AVG >= base - 10:
        grade = 4
    elif AVG >= base - 20:
        grade = 3
    elif AVG >= base - 40:
        grade = 2
    else:
        grade = 1
    print(grade)
    basic.show_number(grade)
input.on_button_pressed(Button.B, Grade)
